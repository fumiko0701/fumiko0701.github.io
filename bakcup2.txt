        function copyText() {
            const escalaText = `${nomeCulto}\n${descricaoCulto}\nPC: ${pc}\nCAM 1: ${cam1}\nCAM 2: ${cam2}\nStorys: ${storys}\nRepórter: ${reporter || 'N/A'}\nFotos: ${fotos}`;
